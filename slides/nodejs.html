  • Fonctionnel

var functToApply = [ Math.abs, Math.floor, function(n) { return n*n; } ];
for(f in functToApply) {
	n = functToApply[f](n);
	console.log('applied function:', functToApply[f], 'result:', n);
}
console.log('Final result:', n);
