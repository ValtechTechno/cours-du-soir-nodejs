D'autres paradigmes

  • Fonctionnel 
    var n = -2.9;
var functionToApply = [ Math.abs, Math.floor, function(n) { return n*n; } ];
for(f in functionToApply) {
	n = functionToApply[f](n);
	console.log('applied function:', functionToApply[f], 'result:', n);
}
console.log('Final result:', n);
